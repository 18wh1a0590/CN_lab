#include <string.h>

	printf("enter the bits:");
    scanf("%s",a);
    int len = strlen(a);
	for(i=0;i<len;i++){
		if(a[i]=='1'){
			count++;
		}
		else{
			count=0;
		}
		b[j]=a[i];
		j++;
		if(count==5 && a[i+1]=='1'){
			b[j]='0';
			j++;
			count=0;
		}
	}

}
